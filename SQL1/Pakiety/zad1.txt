CREATE OR REPLACE PACKAGE konwersja IS
  FUNCTION FAHR_TO_CELS(input NUMBER) RETURN NUMBER;
  FUNCTION CELS_TO_FAHR(input NUMBER) RETURN NUMBER;
END konwersja;

CREATE OR REPLACE PACKAGE BODY konwersja IS
  FUNCTION FAHR_TO_CELS(input NUMBER) RETURN NUMBER IS
  BEGIN
    RETURN  (5/9 * (input - 32)); 
  END FAHR_TO_CELS;
  FUNCTION CELS_TO_FAHR(input NUMBER) RETURN NUMBER IS
  BEGIN
    RETURN  ((9/5 * input) + 32) ; 
  END CELS_TO_FAHR;  
END konwersja;

SELECT konwersja.CELS_TO_FAHR(0) CELSJUSZ FROM DUAL;
